16  package org.yaml.snakeyaml.issues.issue115;

18  import junit.framework.TestCase;
19  
20  import org.yaml.snakeyaml.Yaml;
21  
22  public class ParameterizedJavaBeanTest extends TestCase {
23  
24      public void testAsStandalone() {
25          ParameterizedBean<Integer, String> bean = new ParameterizedBean<Integer, String>();
26          bean.setK(13);
27          bean.setV("ID47");
28          Yaml yaml = new Yaml();
29          String result = yaml.dump(bean);
30          assertEquals("!!org.yaml.snakeyaml.issues.issue115.ParameterizedBean {k: 13, v: ID47}\n",
31                  result);
32          // load
33          @SuppressWarnings("unchecked")
34          ParameterizedBean<Integer, String> beanParsed = (ParameterizedBean<Integer, String>) yaml
35                  .load(result);
36          assertEquals(new Integer(13), beanParsed.getK());
37          assertEquals("ID47", beanParsed.getV());
38      }
39  
40      public void testAsJavaBeanProperty() {
41          Yaml yaml = new Yaml();
42          IssueBean issue = new IssueBean();
43          ParameterizedBean<Integer, String> bean = new ParameterizedBean<Integer, String>();
44          bean.setK(432);
45          bean.setV("Val432");
46          issue.setBean(bean);
47          String result = yaml.dump(issue);
48          assertEquals("!!org.yaml.snakeyaml.issues.issue115.IssueBean\nbean: {k: 432, v: Val432}\n",
49                  result);
50          // load
51          IssueBean issueParsed = (IssueBean) yaml.load(result);
52          assertEquals(new Integer(432), issueParsed.getBean().getK());
53          assertEquals("Val432", issueParsed.getBean().getV());
54      }
